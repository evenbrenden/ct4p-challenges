F :: d -> C(Ld, c)
G :: d -> D(d, Rc)

ψ :: F -> G
f :: x -> y

Ff :: C(Lx, c) -> C(Ly, c)
Gf :: D(x, Rc) -> D(y, Rc)

ψx :: C(Lx, c) -> D(x, Rc)
ψy :: C(Ly, c) -> D(y, Rc)

Gf * ψx :: D(x, Rc) -> D(y, Rc) * C(Lx, c) -> D(x, Rc)
Gf * ψx :: C(Lx, c) -> D(y, Rc)

ψy * Ff :: C(Ly, c) -> D(y, Rc) * C(Lx, c) -> C(Ly, c)
ψy * Ff :: C(Lx, c) -> D(y, Rc)

Gf * ψx = ψy * Ff
